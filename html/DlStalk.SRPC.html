SRPC process handling a request from the client c
    Inductive SRPC_Busy_State (c : Name) :=
    | BWorking : SRPC_Busy_State c
    | BLocked (s : Name) : SRPC_Busy_State c
    .

    #[export] Hint Constructors SRPC_Busy_State : LTS.

Behavioral description of an Single-threaded RPC process
    Inductive SRPC_Busy [c] : SRPC_Busy_State c → Proc → Prop :=
    | SRPC_Busy_Working [P0]
        
        (HReply : ∀ c' v P1, (P0 =(Send (c', R) v)=> P1) → c = c')

        
        (HQuery : ∀ s v P1, (P0 =(Send (s, Q) v)=> P1) → SRPC_Busy (BLocked c s) P1)

        
        (HTau : ∀ P1, (P0 =(Tau)=> P1) → SRPC_Busy (BWorking c) P1)

        
        
        
        (HNoRecv : ∀ h, P0 ≠ PRecv h)

        
        (HNoEnd : P0 ≠ PEnd)
      : SRPC_Busy (BWorking c) P0

    | SRPC_Busy_Locked [s P0]
        
        (HReplyAll : ∀ v, ∃ P1, (P0 =(Recv (s, R) v)=> P1))
        
        (HReplyOnly : ∀ a P1, P0 =(a)=> P1 → ∃ v, a = Recv (s, R) v)
        
        (HRecvR : ∀ a P1, P0 =(a)=> P1 → SRPC_Busy (BWorking c) P1)
      : SRPC_Busy (BLocked c s) P0
    .

    .

    Notation "'Locked'" := (fun c s ⇒ Busy (BLocked c s)) (at level 30).

    Notation "'Locked' c s" := (Busy (BLocked c s)) (at level 200) : type_scope.

Behavioral description of an Single-threaded RPC process
    CoInductive SRPC : SRPC_State → Proc → Prop :=
    | SRPC_Ready [P0]
      
      (HQueryAll : ∀ c v, ∃ P1, (P0 =(Recv (c, Q) v)=> P1))
      
      (HQueryOnly : ∀ a P1, (P0 =(a)=> P1) →
                            ∃ c v,
                              a = Recv (c, Q) v
                              ∧ SRPC (Working c) P1
      )
      : SRPC Ready P0

    | SRPC_Busy_ [P0 c]
        (srpc : SRPC_Busy_State c)
        (HBusy : SRPC_Busy srpc P0)
        (HReply : ∀ v P1, P0 =(Send (c, R) v)=> P1 → SRPC Ready P1)
        (HQuery : ∀ s v P1, P0 =(Send (s, Q) v)=> P1 → SRPC (Locked c s ) P1)
        (HRecv : ∀ s v P1, P0 =(Recv (s, R) v)=> P1 → SRPC (Working c) P1)
        (HTau : ∀ P1, P0 =(Tau)=> P1 → SRPC (Working c) P1)
      : SRPC (Busy srpc) P0
    .

    Definition SRPC_serv srpc S := match S with serv _ P _ ⇒ SRPC srpc P end.

    Definition AnySRPC_serv S := ∃ s, SRPC_serv s S.

    Definition AnySRPC P := ∃ s, SRPC s P.

    #[export] Hint Transparent SRPC_serv AnySRPC AnySRPC_serv : LTS.

    Lemma srpc_inv
      : ∀ (srpc : SRPC_State) (P0 : Proc) (P : Prop),

        (∀ (h : NameTag → option (Val → Proc))
           (hq : Name → Val → Proc),
            (∀ c, match h (c, Q) with
                  | None ⇒ False
                  | Some cont ⇒ ∀ v, cont v = hq c v
                  end
            ) →
            (∀ c v, SRPC (Working c) (hq c v)) →
            (∀ c, h (c, R) = None) →
            P0 = PRecv h →
            srpc = Ready →
            P
        ) →

        (∀ (P1 : Proc) c,
            SRPC (Working c) P1 →
            STau P1 = P0 →
            srpc = Working c →
            P
        ) →

        (∀ (P1 : Proc) c s v,
            SRPC (Locked c s) P1 →
            P0 = PSend (s, Q) v P1 →
            srpc = Working c →
            P
        ) →

        (∀ (P1 : Proc) c v,
           SRPC Ready P1 →
           P0 = PSend (c, R) v P1 →
           srpc = Working c →
           P
        ) →

        (∀ (c s : Name)
           (h : NameTag → option (Val → Proc))
           (hr : Name → Val → Proc),
            (match h (s, R) with
             | None ⇒ False
             | Some cont ⇒ ∀ v, cont v = hr s v
             end
            ) →
            (∀ v, SRPC (Working c) (hr s v)) →
            (∀ n t, n ≠ s → h (n, t) = None) →
            (∀ n, h (n, Q) = None) →
            P0 = PRecv h →
            srpc = Ready →
            P
        ) →

        SRPC srpc P0 → P.
    Abort.

    Ltac2 rec destruct_SRPC (hs : ident) (htr : ident option) :=
      let hs_h := hyp hs in
      match! (Constr.type hs_h) with
      | AnySRPC_serv _ ⇒
          let srpc := Fresh.in_goal @srpc in
          destruct $hs_h as [$srpc $hs];
          destruct_SRPC hs htr
      | AnySRPC _ ⇒
          let srpc := Fresh.in_goal @srpc in
          destruct $hs_h as [$srpc $hs];
          destruct_SRPC hs htr
      | SRPC_serv ?srpc (serv _ _ _) ⇒
          simpl in $hs;
          destruct_SRPC hs htr
      | SRPC_serv ?srpc _ ⇒
          let i := Fresh.in_goal @I in
          let p := Fresh.in_goal @P in
          let o := Fresh.in_goal @O in
          destruct p as [$i $p $o];
          destruct_SRPC hs htr
      | SRPC ?srpc _ ⇒
          let p1 := Fresh.in_goal @P in
          let hqa := Fresh.in_goal @HQueryAll in
          let hqo := Fresh.in_goal @HQueryOnly in
          let c := Fresh.in_goal @c in
          let v := Fresh.in_goal @v in
          let srpcb := Fresh.in_goal @srpc in
          let hb := Fresh.in_goal @HBusy in
          let hr := Fresh.in_goal @HReply in
          let hrc := Fresh.in_goal @HRecv in
          let hq := Fresh.in_goal @HQuery in
          let ht := Fresh.in_goal @HTau in
          let he1 := Fresh.in_goal @HEq1 in
          let he2 := Fresh.in_goal @HEq2 in
          inversion_stable $hs_h
            as [ $p1 $hqa $hqo $he1 $he2 | $p1 $c $srpcb $hb $hr $hq $hrc $ht $he1 $he2]
               >
                 [ ssubst;
                   first
                     [ bs
                     | match htr with
                       | None ⇒ ()
                       | Some htr ⇒
                           let hqo_h := hyp hqo in
                           let htr_h := hyp htr in
                           destruct ($hqo_h _ _ $htr_h) as ($c & $v & $he1 & $hs);
                           clear $hqo
                       end;
                       ssubst
                     ]

                 | ssubst;
                   first
                     [ bs
                     | destruct_SRPC hb htr;
                       Control.enter
                         (fun () ⇒
                            ssubst;
                            match htr with
                            | None ⇒ ()
                            | Some htr ⇒
                                let htr_h := hyp htr in
                                let hr_h := hyp hr in
                                let hrc_h := hyp hrc in
                                let hq_h := hyp hq in
                                let ht_h := hyp ht in
                                first
                                  [ specialize ($hr_h _ _ $htr_h); clear $hq $ht $hrc
                                  | specialize ($hq_h _ _ _ $htr_h); clear $hr $ht $hrc
                                  | specialize ($ht_h _ $htr_h); clear $hr $hq $hrc
                                  | specialize ($hrc_h _ _ _ $htr_h); clear $hr $hq $ht
                                  | ()
                                  ]
                            end
                         )
                     ]
                 ]
      | SRPC_Busy _ _ ⇒
          let p1 := Fresh.in_goal @P in
          let s := Fresh.in_goal @s in
          let hrr := Fresh.in_goal @HRecvR in
          let hsr := Fresh.in_goal @HSendR in
          let hsq := Fresh.in_goal @HSendQ in
          let ht := Fresh.in_goal @HTau in
          let hnr := Fresh.in_goal @HNoRecv in
          let hne := Fresh.in_goal @HNoEnd in
          let hra := Fresh.in_goal @HReplyAll in
          let hro := Fresh.in_goal @HReplyOnly in
          let he1 := Fresh.in_goal @HEq1 in
          let he2 := Fresh.in_goal @HEq2 in
          inversion_stable $hs_h
            as [$p1 $hsr $hsq $ht $hnr $hne $he1 $he2 | $s $p1 $hra $hro $hrr $he1 $he2]
               >
                 [ ssubst;

                   first
                     [ bs
                     | let n := Fresh.in_goal @n in
                       let t := Fresh.in_goal @t in
                       let v := Fresh.in_goal @v in
                       match htr with
                       | None ⇒ ()
                       | Some htr ⇒
                           let htr_h := hyp htr in
                           let a := match! Constr.type htr_h with
                                    | (_ =(?a)=> _) ⇒
                                        if Constr.is_var a then a
                                        else
                                          let av := Fresh.in_goal @a in
                                          remember $a as $av;
                                          hyp av
                                    end in
                           let hq_h := hyp hsq in
                           let hr_h := hyp hsr in
                           let ht_h := hyp ht in
                           destruct $a
                             as [ [$n [|]] $v | [$n $t] $v |];
                           ssubst;
                           try (specialize ($hq_h _ _ _ $htr_h); rename n into s);
                           try (specialize ($hr_h _ _ _ $htr_h); subst $n);
                           try (specialize ($ht_h _ $htr_h));
                           try (inversion $htr_h; bs)
                       end
                     ]

                 | ssubst;
                   first
                     [ bs
                     | let v := Fresh.in_goal @v in
                       let htr_h := match htr with
                                    | None ⇒
                                        let t := Fresh.in_goal @T in
                                        let p := Fresh.in_goal @P in
                                        let hra_h := hyp hra in
                                        destruct ($hra_h some_val) as [$p $t];
                                        hyp t
                                    | Some htr ⇒ hyp htr
                                    end in
                       let hro_h := hyp hro in
                       destruct ($hro_h _ _ $htr_h)
                         as [$v $he1];
                       ssubst
                     ]
                 ]
      end.

    Ltac2 Notation "destruct" "SRPC" hs(ident) ht(opt(ident)) :=
      Control.enter (fun () ⇒ destruct_SRPC hs ht).
End SRPC_DEFS.

SRPC is preserved for processes
  Lemma AnySRPC_invariant : trans_invariant AnySRPC always.

  Proof with eattac.
    intros N0 N1 a T Hsrpc _.
    destruct Hsrpc as [srpc Hsrpc].
    destruct a; destruct srpc; kill Hsrpc.
    all: try (apply HQueryOnly in T; eattac).
    all: try (destruct n as [?[|]]).
    - apply HQuery in T; eattac.
    - hsimpl in H0. kill HBusy.
      + specialize (HReply0 _ _ _ T); subst.
        eapply HReply in T; eattac.
      + apply HReplyOnly in T; attac.
    - hsimpl in H0. kill HBusy.
      + attac.
      + apply HReplyOnly in T; attac.
    - attac.
    - attac.
  Qed.

  #[export] Hint Resolve AnySRPC_invariant : inv.
  #[export] Hint Extern 10 (AnySRPC _) ⇒ solve_invariant : LTS.

  Lemma SRPC_Busy_work_act [P0 : Proc] [c] :
    SRPC_Busy (BWorking c) P0 →
    ∃ P1, (∃ v, P0 =(Send (c, R) v)=> P1) ∨ (P0 =(Tau)=> P1) ∨ (∃ s v, P0 =(Send (s, Q) v)=> P1).

  Proof.
    intros Hsrpc.
    destruct P0; destruct SRPC Hsrpc; ssubst; subst.     - ∃ P0.
      destruct n as [n [|]].
      + right.
        right.
        ∃ n, v.
        constructor.
      + left.
        ∃ v.
        erewrite HSendR.
        constructor.
        constructor.
    - ∃ P0.
      right.
      left.
      constructor.
  Qed.

  Lemma SRPC_busy_reply [P0 P1 P2 path] [c c' v] [s : SRPC_Busy_State c] :
    SRPC (Busy s) P0 →
    (P0 =[path]=> P1) →
    (P1 =(Send (c', R) v)=> P2) →
    Forall (fun a ⇒ match a with
                     | Send (_, t) _ ⇒ t = Q
                     | _ ⇒ True
                     end

      ) path →
    c = c'.

  Proof.
    intros Hsrpc T0 T1 HF.
    generalize dependent P0 c c' v.
    induction path; intros.
    {
      kill T0.
      kill Hsrpc; kill HBusy.
      - attac.
      - eapply HReplyOnly in T1; attac.
    }

    kill T0.
    kill Hsrpc.
    kill HBusy.
    - eattac.
      kill T2; attac.
      destruct n; attac.
    - hsimpl in HF.
      destruct a.
      + destruct n; attac.
      + destruct n as [?[|]]. 2: eattac.
        apply HReplyOnly in T2; attac.
      + apply HReplyOnly in T2; attac.
  Qed.

  Lemma SRPC_busy_reply_exists [P0] [c] [s : SRPC_Busy_State c] :
    SRPC (Busy s) P0 →
    ∃ path P1 P2 v,
      (P0 =[path]=> P1)
      ∧ (P1 =(Send (c, R) v)=> P2)
      ∧ Forall (fun a ⇒ match a with
                          | Send (_, t) _ ⇒ t = Q
                          | _ ⇒ True
                          end) path.

  Proof with eattac.
    intros Hsrpc.
    assert (∀ a P', P0 =(a)=> P' → AnySRPC P') as HPres by eattac.

    kill Hsrpc.
    ltac1:(dependent destruction H0).

    ltac1:(dependent induction HBusy); intros.
    - assert (SRPC_Busy (BWorking c) P0) as Hsrpc by (constructor; eattac).
      specialize (SRPC_Busy_work_act Hsrpc) as [P1 [[v T]|[T|[s [v T]]]]].
      + ∃ [], P0, P1, v...
      + specialize (H0 P1 T).
        specialize (HTau0 _ T).
        specialize (HTau _ T).
        kill HTau0.
        destruct H0 as (path & P2 & P3 & v & T1 & T2 & HF); eauto with LTS.
        ∃ (Tau :: path), P2, P3, v; eauto with LTS.
      + specialize (H s v P1 T).
        specialize (HQuery _ _ _ T).

    consider (∃ c0, SRPC_serv (Locked c0 n0) S0) by eauto using lock_SRPC_Locked_serv with LTS.
    consider (∃ c1, SRPC_serv (Locked c1 n1) S1) by eauto using lock_SRPC_Locked_serv with LTS.
    destruct S0, S1; simpl in ×.
    consider (_ =(a)=> _);
      try (consider (Locked c0 n0 = Locked c1 n1) by eattac).
    eauto using SRPC_no_relock with LTS.
  Qed.

  Lemma SRPC_send_lock [L a P0 P1] :
    AnySRPC P0 →
    proc_lock L P1 →
    (P0 =(a)=> P1) →
    match a with Send (_, t) _ ⇒ t = Q | _ ⇒ False end.

  Proof with (eauto with LTS).
    intros Hsrpc0 HL1 T.

    assert (AnySRPC P1) as Hsrpc1 by attac.

    specialize (SRPC_get_lock Hsrpc1 HL1) as [s HL1s].

    apply (lock_SRPC_Locked Hsrpc1) in HL1s as [c Hsrpc1_L].

    clear Hsrpc1.

      ltac1:(dependent destruction H1).

      kill HBusy0.

      kill T.

      kill T.

  Lemma SRPC_Decisive [P] :
    AnySRPC P →
    Decisive P.

  Proof.
    generalize dependent P.
    ltac1:(cofix C).
    intros P Hsrpc_p.

    destruct P.
    - destruct Hsrpc_p as [srpc Hsrpc_p];
        ltac1:(dependent destruction Hsrpc_p) >
                [|ltac1:(dependent destruction HBusy)].
      + specialize (HQueryAll some_name some_val) as [P1 T].
        kill T.
      + bs.
      + specialize (HReplyAll some_val) as [P1 T].
        kill T.
    - constructor; eattac.

    - constructor; intros;
        destruct Hsrpc_p as [srpc Hsrpc_p];
        ltac1:(dependent destruction Hsrpc_p) >
                [|ltac1:(dependent destruction HBusy)].
      + split; intros.
        × destruct (handle (m, R)) eqn:HH; auto.
          assert (PRecv handle =(Recv (m, R) some_val)=> p some_val) as T by attac.
          apply HQueryOnly in T as (c & v & HEq & _).
          bs.
        × assert (PRecv handle =(Recv (n, Q) v)=> P v) as T by attac.
          apply HQueryOnly in T as (c & _ & _ & Hsrpc_n).
          apply C; eattac.
      + assert (PRecv handle =(Recv (n, Q) some_val)=> P some_val) as T by attac.
        kill T.
      + assert (PRecv handle =(Recv (n, Q) some_val)=> P some_val) as T by attac.
        apply HReplyOnly in T as (v & HEq).
        bs.
      + assert (PRecv handle =(Recv (n, R) some_val)=> P some_val) as T by attac.
        apply HQueryOnly in T as (c & v & HEq & _).
        bs.
      + assert (PRecv handle =(Recv (n, R) some_val)=> P some_val) as T by attac.
        kill T.
      + split; intros.
        × destruct (handle (m, Q)) eqn:HH; auto.
          assert (PRecv handle =(Recv (m, Q) some_val)=> p some_val) as T by attac.
          apply HReplyOnly in T as (v & HEq).
          bs.
        × assert (PRecv handle =(Recv (n, R) v)=> P v) as T by attac.
          apply HRecv in T as Hsrpc_n.
          apply C; eattac.
    - constructor.
      assert (STau P =(Tau)=> P) as T by attac.
      assert (AnySRPC P) by eattac.
      attac.
  Qed.

    AnySRPC_serv S →
    Decisive_serv S.
  Proof. intros. destruct S; eattac. Qed.

  Lemma SRPC_tau_no_lock_r [S0 S1 L] :
    AnySRPC_serv S0 →
    (S0 =(Tau)=> S1) →
    ¬ serv_lock L S1.

  Proof.
    intros; intros ?.
    consider (∃ s, serv_lock [s] S1) by eauto using SRPC_serv_get_lock with LTS.
    consider (∃ c, SRPC_serv (Locked c s) S1) by eauto using lock_SRPC_Locked_serv with LTS.
    consider (_ =(_)=> _); consider (serv_lock _ _); doubt; simpl in ×.
    - destruct n as [n [|]].
      + assert (SRPC (Working n) P1) by eauto using SRPC_inv_recv_Q_r.
        bs (Working n = Locked c s) by attac.
      + consider (∃ c', SRPC (Working c') P1) by eauto using SRPC_inv_recv_R_r.
        bs (Working c' = Locked c s) by attac.
    - consider (∃ c', SRPC (Working c') P1) by eauto using SRPC_inv_tau_r.
      bs (Working c' = Locked c s) by attac.
  Qed.
