Module Paper.
  Definition serv_lock (n : Name) (S : Serv) :=
    (∀ v E, ¬ Deq (n, R) v (serv_i S) E) ∧ (∃ c, SRPC (Locked c n) (serv_p S)) ∧ (serv_o S = []).

  Definition dead_set (DS : list Name) N :=
    DS ≠ [] ∧ (∀ n0, In n0 DS → ∃ n1, lock N n0 n1 ∧ In n1 DS).

  Inductive SRPCI : SRPC_State → Proc → Prop :=
  | SRPCI_R h :
    (∀ c t, h (c, t) ≠ None → t = Q) →
    (∀ c v, ∃ cont, h (c, Q) = Some cont ∧ SRPCI (Working c) (cont v)) →
    SRPCI Ready (PRecv h)
  | SRPCI_WR c v P : SRPCI (Working c) (PSend (c, R) v P)
  | SRPCI_WT c P : SRPCI (Working c) P → SRPCI (Working c) (STau P)
  | SRPCI_WQ c s v P : SRPCI (Locked c s) P → SRPCI (Working c) (PSend (s, Q) v P)
  | SRPCI_L c s h cont :
    (∀ nc, h nc ≠ None → nc = (s, R)) →
    (h (s, R) = Some cont) →
    (∀ v, SRPCI (Working c) (cont v)) →
    SRPCI (Locked c s) (PRecv h)
  .

  CoInductive SRPCC : SRPC_State → Proc → Prop :=
  | SRPCC_R P0 : (∀ c v P1, P0 =(Recv (c, Q) v)=> P1 → SRPCC (Working c) P1) → SRPCI Ready P0 → SRPCC Ready P0
  | SRPCC_W c P0 :
    (∀ v P1, P0 =(Send (c, R) v)=> P1 → SRPCC Ready P1) →
    (∀ P1, P0 =(Tau)=> P1 → SRPCC (Working c) P1) →
    (∀ s v P1, P0 =(Send (s, Q) v)=> P1 → SRPCC (Locked c s) P1) →
    SRPCI (Working c) P0 → SRPCC (Working c) P0
  | SRPCC_L c s P0 : (∀ v P1, P0 =(Recv (s, R) v)=> P1 → SRPCC (Working c) P1) → SRPCI (Locked c s) P0 → SRPCC (Locked c s) P0
  .

  Definition client (n : Name) (S : Serv) :=
    (∃ v, In (n, Q, v) (serv_i S) ∨ In (n, R, v) (serv_o S))
    ∨ SRPCC (Working n) (serv_p S) ∨ (∃ s, SRPCC (Locked n s) (serv_p S)).

End Paper.

Fact serv_lock_eq : ∀ S n, AnySRPC_serv S → serv_lock [n] S ↔ Paper.serv_lock n S.
Proof.
  split; intros.
  - consider (serv_lock _ _).
    repeat split; repeat (intros ?); simpl in ×.
    + bs (In (n, R, v) &I).
    + eauto using lock_SRPC_Locked with LTS.
  - destruct S.
    consider (Paper.serv_lock _ _).
    hsimpl in ×.
    repeat split; repeat (intros ?); simpl in *; eauto with LTS.
    assert (∃ v' E, Deq (n0, R) v' serv_i0 E) by eauto using In_Deq.
    hsimpl in ×.
    bs.
Qed.

Fact dead_set_eq : ∀ N DS, SRPC_net N → dead_set DS N ↔ Paper.dead_set DS N.

Proof.
  unfold Paper.dead_set.
  split; intros; repeat constructor; intros.
  - attac.
  - consider (dead_set _ _).
    consider (∃ L, lock_set N L n0 ∧ incl L DS).
    unfold lock_set, lock in ×.
    consider (∃ n1, serv_lock [n1] (NetMod.get n0 N)) by eauto with LTS.
    ∃ n1.
    eattac.
  - attac.
  - hsimpl in ×.
    consider (∃ n1, lock N n n1 ∧ _) by eauto.
    ∃ [n1].
    unfold incl, lock in ×.
    attac.
Qed.

Inductive SRPC_measure : Proc → Prop :=
| ms_reply n v P : SRPC_measure (PSend (n, R) v P)
| ms_tau P : SRPC_measure P → SRPC_measure (STau P)
| ms_query n v P : SRPC_measure P → SRPC_measure (PSend (n, Q) v P)
| ms_recv h : (∀ v nc cont, h nc = Some cont → SRPC_measure (cont v)) → SRPC_measure (PRecv h)
.

Lemma SRPCI_measure : ∀ srpc P, srpc ≠ Ready → Paper.SRPCI srpc P → SRPC_measure P.
Proof.
  intros.
  induction H0; intros; constructor; eattac.
  specialize (H0 nc ltac:(attac)); attac.
  rewrite H4 in H1.
  attac.
Qed.

Lemma SRPCC_SRPCI : ∀ srpc P, Paper.SRPCC srpc P → Paper.SRPCI srpc P.
  intros. kill H.
Qed.

Lemma SRPCC_measure : ∀ srpc P, srpc ≠ Ready → Paper.SRPCC srpc P → SRPC_measure P.
  intros.
  eauto using SRPCC_SRPCI, SRPCI_measure.
Qed.
